from llama_cpp import Llama def process_query(query): llm = Llama( model_path="model.gguf", n_ctx=2048, verbose=False ) response = llm(query, max_tokens=200) return response["choices"][0]["text"].strip() if __name__ == "__main__": user_query = input("Enter your query: ") answer = process_query(user_query) print("\nAnswer:") print(answer)
